type colour {
  return this[0] === "#";
  //A colour is just any sting starting with a hashtag.
}

class Person (obj: options) {
  using options; //Import options namespace.
  public str: name = name | "A person";

  //Only assign if exists in the current scope/namespace.
  public int: age ?= age;
  public colour: hairColour ?= hairColour;

  greet (str: greeting) {
    print(`${name} says ${greeting}!`);
  }
}

const peter = new Person([
  name = "Peter",
  age = 21,
  hairColour = "#F99999"
]);

const [Person]: people = [
  peter,
  new Person(),
  new Person()
];

for (person in people) {
  person.greet("hi");
}
